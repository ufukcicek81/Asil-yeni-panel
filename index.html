function updateDovizBant() { var bantItems = []; if (liveDoviz && liveDoviz.usdAlis) { var dovizListesi = [ { sym:'🇺🇸 USD', id:'usd' }, { sym:'🇪🇺 EUR', id:'eur' }, { sym:'🇬🇧 GBP', id:'gbp' }, { sym:'🇸🇦 SAR', id:'sar' }, { sym:'🇨🇭 CHF', id:'chf' } ]; dovizListesi.forEach(function(d) { var duzA = parseFloat((settings.dovizDuz && settings.dovizDuz[d.id + 'A']) || 0); var duzS = parseFloat((settings.dovizDuz && settings.dovizDuz[d.id + 'S']) || 0); var alis = liveDoviz[d.id + 'Alis'] ? (liveDoviz[d.id + 'Alis'] + duzA).toFixed(2) : null; var satis = liveDoviz[d.id + 'Satis'] ? (liveDoviz[d.id + 'Satis'] + duzS).toFixed(2) : null; if (alis && satis) { bantItems.push({ sym: d.sym, alis: alis, satis: satis }); } }); } if (!bantItems.length) { document.getElementById('dovizBant').innerHTML = '⏳ Döviz yükleniyor...'; return; } var html = ''; bantItems.forEach(function(it) { html += '' + it.sym + ' ' + it.alis + '' + '/' + '' + it.satis + '' + ' ₺'; }); document.getElementById('dovizBant').innerHTML = html + html; }
